        function simulateDataUpdate() {

            platformsData = platformsData.map(data => {

                const fluctuation = (Math.random() * 0.2 - 0.1);

                // Ensure APY stays within a reasonable range (e.g., 0.5% to 15%)

            platformsData.sort((a, b) => b.apy - a.apy);

        }
